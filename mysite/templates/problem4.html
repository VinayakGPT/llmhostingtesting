4. Smallest Prime: Type-1

Problem Statement:
Given a positive integer N, print the number p, such that p is the smallest prime number greater than or equal to N.
Example:
Input: 11
Output: 11
Input: 22
Output: 23
Input: 1
Output: 2

Incorrect Solution:
#include <stdio.h>
int check_prime(int num) {
    int j = 2;
    while (j <= num - 1) {
        if (num % j == 0)
            return 0;
        j = j + 1;
    }
    return 1;
}
int main() {
    int N, i;
    scanf("%d", &N);
    i = N;
    while (i >= N) {
        if (check_prime(i))
            break;
        i = i + 1;
    }
    printf("%d", i);
    return 0;
}
