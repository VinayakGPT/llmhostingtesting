5. Smallest Prime: Type-2

Problem Statement:
Given a positive integer N, print the number p, such that p is the smallest prime number greater than or equal to N.
Example:
Input: 11
Output: 11
Input: 22 
Output: 23
Input: 1
Output: 2

Incorrect Solution:
#include <stdio.h>
#include <math.h>
int check_prime(int num)
{
    int num_divisors=0;
    if(num==1)
        return 0;

    if (num==2||num==3)
        return 1;
    else
    {
        for(int i=2;i<(int)sqrt(num);i++) 
            if(num%i==0)
                num_divisors++;
        if(num_divisors == 0) 
            return 1;

        else
            return 0;
    }
}
int main()
{
    int a,o;
    scanf("%d",&a);
    do
    {
        o=check_prime(a);
        if(o==1)
            printf("%d\n",a);
        else
            a++;    
    }
    while(o!=1);
    return 0;
}
